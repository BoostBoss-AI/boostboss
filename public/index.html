Integrate the SDK → 
// Add Boost Boss in one line
import { BoostBoss } from '@boostbossai/lumi-sdk';

const bb = new BoostBoss({
  apiKey: 'bb_sk_live_***',
  format: 'native'
});

const ad = await bb.fetch({
  context: userMessage,
  audience: 'ai-native'
});

render(ad); // 💰
